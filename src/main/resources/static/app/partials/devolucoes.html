// devolucoes.js - Lógica para a página de histórico de devoluções (async function() { console.log("A executar o script da página de devoluções..."); const corpoTabela = document.getElementById('corpo-tabela-devolucoes'); if (!corpoTabela) { console.error("Elemento da tabela de devoluções não encontrado!"); return; } // Função para buscar e renderizar as devoluções na tabela async function carregarDevolucoes(page = 0, size = 10) { corpoTabela.innerHTML = 'A carregar...'; try { const params = { page, size, sort: 'dataDevolucao,desc' }; // Usa a apiClient para buscar a lista paginada de devoluções const response = await apiClient.get('/devolucoes', { params }); const devolucoes = response.data.content; corpoTabela.innerHTML = ''; // Limpa a tabela if (devolucoes.length === 0) { corpoTabela.innerHTML = 'Nenhuma devolução registada.'; return; } devolucoes.forEach(dev => { const tr = document.createElement('tr'); tr.innerHTML = ` ${dev.id} ${dev.solicitacaoId} ${dev.nomeEquipamento} ${dev.quantidadeDevolvida} ${new Date(dev.dataDevolucao).toLocaleString('pt-BR')} ${dev.observacao || ''} `; corpoTabela.appendChild(tr); }); // Adicionar lógica de paginação aqui no futuro } catch (error) { console.error("Erro ao carregar devoluções:", error); showToast("Não foi possível carregar o histórico de devoluções.", "Erro", true); corpoTabela.innerHTML = 'Falha ao carregar devoluções.'; } } // --- Inicialização --- carregarDevolucoes(); })();
